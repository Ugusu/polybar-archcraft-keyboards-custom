> sudo cp -r ./Backgrounds /usr/share/sddm/themes/archcraft/

> cp -r ./picom.conf ~/.config/
> cp -r ./config.rasi ~/.config/rofi/

> betterlockscreen -u /usr/share/backgrounds/anime/
